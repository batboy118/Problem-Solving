vector<bool> solution(vector<string> infos, vector<string> actions) {
	vector<bool> answer;
	vector <string> str;
	string token;
	bool login = false;
	bool order = false;
	char delimiter = ' ';
	int temp;

int Usercnt = infos.size();
for (int i = 0; i<Usercnt; i++) {
	stringstream ss(infos[i]);
	while (getline(ss, token, delimiter)) {
		str.push_back(token);
	}
	Users[i].id = str[0];
	Users[i].password = str[1];
	str.clear();
}
for (int i = 0; i<actions.size(); i++) {
	stringstream ss(actions[i]);
	while (getline(ss, token, delimiter)) {
		str.push_back(token);
	}
	if (str[0] == "ORDER") {
		if (order) {
			answer.push_back(true);
			order = false;
		}
		else {
			answer.push_back(false);
		}
	}
	else if (str[0] == "ADD") {
		if (login) {
			answer.push_back(true);
			order = true;
		}
		else {
			answer.push_back(false);
		}
	}
	else if (str[0] == "LOGIN") {
		if (login) {
			answer.push_back(false);
		}
		else{
			for (int j = 0; j < Usercnt; j++) {
				if (Users[j].id == str[1]) {
					if (Users[j].password == str[2]) {
						answer.push_back(true);
						login = true;
					}
					else {
						answer.push_back(false);
					}
				}
			}
		}
	}
	str.clear();
}
return answer;

}
